    <script>
        const API_URL = 'https://gist.githubusercontent.com/assrabagi-afk/896d7730bbed8f1da5665a7913a89f2a/raw/8ae161d4edc2a04f792673ffc8d6423c6a0343e9/gistfile1.txt';

        async function handleLogin() {
            const nisnInput = document.getElementById('nisnInput').value;
            const dobInput = document.getElementById('dobInput').value;
            const errorMsg = document.getElementById('errorMsg');
            const loginBtn = document.getElementById('loginBtn');

            if (!nisnInput || !dobInput) {
                showError("Harap isi semua kolom!");
                return;
            }

            loginBtn.disabled = true;
            loginBtn.innerText = "Memproses...";

            try {
                const response = await fetch(API_URL);
                const rawData = await response.text();
                
                // Parsing CSV-like data
                const lines = rawData.trim().split('\n');
                const students = lines.slice(1).map(line => {
                    const [dob, nisn, nama, mat, indo, rata] = line.split(';');
                    return { dob, nisn, nama, mat, indo, rata };
                });

                const student = students.find(s => s.nisn === nisnInput && s.dob === dobInput);

                if (student) {
                    showResult(student);
                } else {
                    showError("Data tidak ditemukan. Cek kembali NISN dan Tanggal Lahir.");
                }
            } catch (error) {
                showError("Gagal mengambil data. Coba lagi nanti.");
            } finally {
                loginBtn.disabled = false;
                loginBtn.innerText = "Cek Hasil Sekarang";
            }
        }

        function showError(msg) {
            const errorMsg = document.getElementById('errorMsg');
            errorMsg.innerText = msg;
            errorMsg.classList.remove('hidden');
            errorMsg.classList.add('animate__animated', 'animate__shakeX');
        }

        function showResult(data) {
            document.getElementById('loginSection').classList.add('hidden');
            document.getElementById('resultSection').classList.remove('hidden');
            
            document.getElementById('resNama').innerText = data.nama;
            document.getElementById('resNisn').innerText = "NISN: " + data.nisn;
            document.getElementById('resMat').innerText = data.mat;
            document.getElementById('resIndo').innerText = data.indo;
            document.getElementById('resRata').innerText = data.rata;
        }
    </script>
